```c++
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int left = 0;
        int right = 0;
        int ans = 0;
        int count = 0;
        int index = 0;//记录左边界位置

        while(s[right]){

            for(left = index;left < right;left++){
                if(s[left] == s[right]){
                    index++;
                    count = 0;
                    right = index;
                    break;
                }
            }
            right++;
            count++;
            ans = max(count,ans);
        }
        return ans;
    }
};
```
